git 설정

2016-12-16
#git  #개발환경  
git config
git config --global user.name "name"
git config --global user.email "email address"

사용중인 gitconfig 기본 설정(~/.gitconfig)

[user]
	name = name
	email = email address
[core]
	autocrlf = input
	eol = native
[color]
	ui = auto
[alias]
	tree = log --oneline --decorate --all --graph

git 설정

December 16, 2016 in index 
git config
git config --global user.name "name"
git config --global user.email "email address"

사용중인 gitconfig 기본 설정(~/.gitconfig)

[user]
	name = name
	email = email address
[core]
	autocrlf = input
	eol = native
[color]
	ui = auto
[alias]
	tree = log --oneline --decorate --all --graph
